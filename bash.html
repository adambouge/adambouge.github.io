Simple IP scanner

It finds active IPs on the local network

#!/bin/bash

# We will test the last part of the ip
prefix="192.168.0."

# Test last part of the ip from 1 to 255
for i in {1..255}
do
    # Try to ping once the ip
    ping -c 1 $prefix$i > /dev/null
    
    # If ping is success
    if (($? == 0))
    then
        echo $prefix$i is online
    fi
done

Simple port scanner

It finds the open ports of an IP

#!/bin/bash

# Ip is given as an argument
ip=$1
    
# Test ports from 1 to 100 (max: 65535)
for port in {1..100}
do
     # Use nc to test the port
    nc -z -w 1 $ip $port
        
    # If nc returns success
    if (($? == 0))
    then
        echo port $port is open
    fi
done
